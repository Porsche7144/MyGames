// HLSL

cbuffer HDataCB
{
	float4 vColor : packoffset(c0.x);
	float  fTime  : packoffset(c1.x);
};
struct VS_OUTPUT
{
	float4 p : SV_POSITION;
	float4 c : COLOR0;
};
struct PS_OUTPUT
{
	float4 c : SV_TARGET;
};
PS_OUTPUT PS( VS_OUTPUT vIn)
{
	PS_OUTPUT vOut;
	vOut.c = lerp(vIn.c,vColor, fTime);
	return vOut;
}
