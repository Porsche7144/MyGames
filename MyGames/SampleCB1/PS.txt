// HLSL
Texture2D g_txDiffuse;  // : resigter(t0)  입력값 없으면 자동으로 t0
SamplerState g_Sample;  // : register(s0); 입력값 없으면 자동으로 s0
cbuffer HDataCB
{
	float4 vColor : packoffset(c0.x);
	float  fTime  : packoffset(c1.x);
};

struct VS_OUTPUT
{
	float4 p : SV_POSITION;
	float3 n : NORMAL;
	float4 c : COLOR0;
	float2 t : TEXCOORD0;
};

struct PS_OUTPUT
{
	float4 c : SV_TARGET;
};

PS_OUTPUT PS( VS_OUTPUT vIn)
{
	PS_OUTPUT vOut;
	float4 vTexture = g_txDiffuse.Sample( g_Sample, vIn.t);
	vOut.c = vTexture;
	return vOut;
}
