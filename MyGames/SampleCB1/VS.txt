cbuffer HDataCB
{
	float4 vColor : packoffset(c0.x);
	float  fTime  : packoffset(c1.x);
	float  fGameTime  : packoffset(c1.y);
};
struct VS_INPUT
{
	float3 p : POSITION;
	float3 n : NORMAL;
	float4 c : COLOR;
	float2 t : TEXTURE;
};
struct VS_OUTPUT
{
	float4 p : SV_POSITION;
	float3 n : NORMAL;
	float4 c : COLOR0;
	float2 t : TEXCOORD0;
};
VS_OUTPUT VS( VS_INPUT vIn)
{
	VS_OUTPUT vOut;
	vOut.p = float4(vIn.p,1.0f);
	vOut.n = vIn.n;
	vOut.c = vIn.c;
	vOut.t = vIn.t;

	return vOut;
}
